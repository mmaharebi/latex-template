\section{Enhanced Math Commands}

The template includes many mathematical shortcuts for common operations.

\subsection{Derivatives}

\subsubsection{Partial Derivatives}

First-order partial derivative:
\[
\pd{f}{x} = \frac{\partial f}{\partial x}
\]

Second-order partial derivative:
\[
\pdd{f}{x} = \frac{\partial^2 f}{\partial x^2}
\]

Mixed partial derivative:
\[
\pdc{f}{x}{y} = \frac{\partial^2 f}{\partial x \partial y}
\]

Partial time derivative (common in physics):
\[
\pdt{u} = \frac{\partial u}{\partial t}
\]

\subsubsection{Total Derivatives}

First derivative:
\[
\td{s}{t} = \frac{ds}{dt}
\]

Second derivative:
\[
\tdd{x}{t} = \frac{d^2x}{dt^2} = \ddot{x}
\]

\subsection{Vector Calculus Operators}

\subsubsection{Gradient}
\[
\grad f = \vnabla f = \pd{f}{x}\xhat + \pd{f}{y}\yhat + \pd{f}{z}\zhat
\]

Example with electric potential:
\[
\vect{E} = -\grad V
\]

\subsubsection{Divergence}
\[
\Div \vect{F} = \vnabla \cdot \vect{F} = \pd{F_x}{x} + \pd{F_y}{y} + \pd{F_z}{z}
\]

Gauss's law:
\[
\Div \vect{E} = \frac{\rho}{\epsilon_0}
\]

\subsubsection{Curl}
\[
\curl \vect{F} = \vnabla \times \vect{F}
\]

Faraday's law:
\[
\curl \vect{E} = -\pdt{\vect{B}}
\]

\subsection{Vector Operations}

\subsubsection{Dot Product}
\[
\vect{a} \vdot \vect{b} = |\vect{a}||\vect{b}|\cos\theta
\]

\subsubsection{Vector Projection}
Projection of $\vect{b}$ onto $\vect{a}$:
\[
\text{proj}_{\vect{a}}\vect{b} = \vproj{a}{b} = \frac{\vect{a} \vdot \vect{b}}{|\vect{a}|^2}\vect{a}
\]

\subsubsection{Angle Between Vectors}
\[
\theta = \vang{a}{b} = \arccos\left(\frac{\vect{a} \vdot \vect{b}}{|\vect{a}||\vect{b}|}\right)
\]

\subsection{Matrix Operations}

\subsubsection{Trace}
\[
\trace(\matr{A}) = \sum_{i=1}^n A_{ii}
\]

\subsubsection{Rank}
\[
\rank(\matr{A}) = \text{number of linearly independent columns}
\]

\subsection{Utilities}

\subsubsection{Absolute Value and Norm}
\[
\abs{x} = |x|, \quad \norm{\vect{v}} = \sqrt{\vect{v} \vdot \vect{v}}
\]

\subsubsection{Probability}
\[
\Prob{A \cap B} = \Prob{A} \cdot \Prob{B|A}
\]

\subsubsection{Defined As}
\[
f(x) \eqdef x^2 + 2x + 1
\]

\subsection{Constants}

The template provides common constants:
\begin{align*}
\e &= 2.71828\ldots \quad \text{(Euler's number)} \\
\imag &= \sqrt{-1} \quad \text{(imaginary unit)} \\
\dif &\quad \text{(differential, e.g., } \int f(x) \dif x \text{)}
\end{align*}

Example with Euler's formula:
\[
\e^{\imag\theta} = \cos\theta + \imag\sin\theta
\]
