\newglossaryentry{wavelength}{
  name=wavelength,
  description={The distance between successive peaks of a wave, 
    typically denoted by $\lambda$}
}

\newglossaryentry{frequency}{
  name=frequency,
  description={The number of wave cycles per unit time, 
    measured in Hertz (Hz)}
}

\newglossaryentry{radar}{
  name=radar,
  description={Radio Detection and Ranging, 
    a system that uses electromagnetic waves to detect objects}
}
